const express = require("express"); const mongoose = require("mongoose"); const cors = require("cors"); const app = express(); app.use(cors()); app.use(express.json()); // MongoDB Connection mongoose.connect("mongodb://127.0.0.1:27017/ecommerce"); // Product Schema const productSchema = new mongoose.Schema({ name: String, price: Number, image: String, sizes: [String], stock: Number }); const Product = mongoose.model("Product", productSchema); // Cart Schema const cartSchema = new mongoose.Schema({ productId: String, qty: Number, size: String }); const Cart = mongoose.model("Cart", cartSchema); // Routes app.get("/products", async (req, res) => { const products = await Product.find(); res.json(products); }); app.post("/products", async (req, res) => { const newProduct = new Product(req.body); await newProduct.save(); res.json(newProduct); }); app.post("/cart", async (req, res) => { const item = new Cart(req.body); await item.save(); res.json(item); }); app.get("/cart", async (req, res) => { const items = await Cart.find(); res.json(items); }); app.listen(5000, () => console.log("Backend running on port 5000"));
